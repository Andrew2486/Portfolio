            for (int i = 0; i < line1Split.Length; i++)

                num1[i] = double.Parse(line1Split[i]);
            }


            WriteLine("Please enter the second line of numbers same rules as the first line");
            string line2 = ReadLine();
            string[] line2Split = line2.Split(',', '.', ' ');
            int maxLine2 = line2Split.Length;

            for (int e = 0; e < line2Split.Length; e++)
            {
                num2[e] = double.Parse(line1Split[e]);
            }

            foreach (int a in num1)
            {
                if (num2.Contains(a))
                {
                    count++;

        }
